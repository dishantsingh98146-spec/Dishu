<script>
/* ===== TEXT SETUP ===== */
document.getElementById("name").innerText =
  getComputedStyle(document.documentElement).getPropertyValue('--name').replace(/"/g,'');

document.getElementById("msg").innerText =
  getComputedStyle(document.documentElement).getPropertyValue('--message').replace(/"/g,'');

/* ===== FIREWORKS ===== */
const fw = document.getElementById("fireworks");
const fctx = fw.getContext("2d");
fw.width = innerWidth;
fw.height = innerHeight;

let particles = [];
const colors = ["red","yellow","pink","cyan","lime","orange","violet"];

function firework() {
  const x = Math.random()*fw.width;
  const y = Math.random()*fw.height/2;
  for(let i=0;i<50;i++){
    particles.push({
      x,y,
      dx:(Math.random()-0.5)*6,
      dy:(Math.random()-0.5)*6,
      life:60,
      c:colors[Math.floor(Math.random()*colors.length)]
    });
  }
}

function animateFW(){
  fctx.clearRect(0,0,fw.width,fw.height);
  particles.forEach((p,i)=>{
    fctx.fillStyle=p.c;
    fctx.fillRect(p.x,p.y,3,3);
    p.x+=p.dx;
    p.y+=p.dy;
    p.life--;
    if(p.life<=0) particles.splice(i,1);
  });
  requestAnimationFrame(animateFW);
}

setInterval(firework,700);
animateFW();

/* ===== CONFETTI ===== */
const conf = document.getElementById("confetti");
const cctx = conf.getContext("2d");
conf.width = innerWidth;
conf.height = innerHeight;

let confetti = [];

function blastConfetti(){
  for(let i=0;i<150;i++){
    confetti.push({
      x:Math.random()*conf.width,
      y:-10,
      s:Math.random()*6+4,
      dy:Math.random()*4+2,
      c:colors[Math.floor(Math.random()*colors.length)]
    });
  }
}

function animateConfetti(){
  cctx.clearRect(0,0,conf.width,conf.height);
  confetti.forEach((c,i)=>{
    cctx.fillStyle=c.c;
    cctx.fillRect(c.x,c.y,c.s,c.s);
    c.y+=c.dy;
    if(c.y>conf.height) confetti.splice(i,1);
  });
  requestAnimationFrame(animateConfetti);
}
animateConfetti();

  });
}
